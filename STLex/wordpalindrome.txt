bool IsWordPalindrome(string text)
{
	text.erase(remove_if(text.begin(),text.end(),[](char c){return !isalpha(c)&&!isspace(c);}),text.end());
	transform(text.begin(),text.end(),text.begin(),tolower);

	stringstream tokenizer(text);
	vector<string> tokens;
	tokens.insert(tokens.begin(),
		istream_iterator<string>(tokenizer),
		istream_iterator<string>());
	return equal(tokens.begin(),tokens.begin()+tokens.size()/2,tokens.rbegin());
}
