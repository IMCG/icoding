int used[256]={0};
int match_sum(const string init,string dest )
{
	int count=0;
	for(int i=0;i<4;i++)
	{
		//if(!used[init[i]])
			++used[init[i]];
	}
	for(int i=0;i<4;i++)
	{
		if(used[dest[i]]){
			count++;
			used[dest[i]]--;
		}
	}
	return count;
}
