#include<HE\simpio.h>
#include<iostream>
#include<fstream>
#include<iomanip>
#include<tchar.h>
using namespace std;
const int NUM_LINES=4;
const int NUM_COLUMNS=3;
const int COLUME_WIDTH=18;
void PrintTableBody()
{
	ifstream input(_TEXT("E:\\1.txt"));
	for(int i=0;i<NUM_LINES;++i)
	{
		int intvalue;
		double doublevalue;
		input>>intvalue>>doublevalue;

		cout<<setw(COLUME_WIDTH)<<(i+1)<<"|";
		cout<<setw(COLUME_WIDTH)<<intvalue<<"|";
		cout<<setw(COLUME_WIDTH)<<doublevalue<<"|"<<endl;
	}
}
void PrintTableHeader()
{
	for(int column=0;column<NUM_COLUMNS-1;++column)
	{
		for(int i=0;i<COLUME_WIDTH;++i)
		{
			cout<<'-';
		}
		cout<<"-+-";
	}

	for(int i=0;i<COLUME_WIDTH;++i)
		cout<<'-';
	cout<<endl;
}
int main()
{
	PrintTableHeader();
	PrintTableBody();
	
}
